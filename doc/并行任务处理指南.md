# ThreadPoolExecutor 并行任务处理 MVP 指南

## 🎯 核心概念

`ThreadPoolExecutor` 是 Python 标准库 `concurrent.futures` 提供的线程池执行器，用于并发执行任务。

**适用场景：**
- ✅ I/O 密集型任务（网络请求、文件读写、数据库查询）
- ✅ 需要控制并发数量的批处理任务
- ❌ CPU 密集型任务（应使用 `ProcessPoolExecutor`）

---

## 🚀 3 分钟快速开始

### 步骤 1: 基本用法

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
import time

def process_item(item):
    """处理单个任务"""
    time.sleep(0.5)  # 模拟耗时操作
    return f"Processed: {item}"

# 准备数据
items = [1, 2, 3, 4, 5]

# 使用线程池执行
with ThreadPoolExecutor(max_workers=3) as executor:
    # 提交所有任务
    futures = [executor.submit(process_item, item) for item in items]
    
    # 获取结果（按完成顺序）
    for future in as_completed(futures):
        result = future.result()
        print(result)
```

**输出：**
```
Processed: 2
Processed: 1
Processed: 3
Processed: 4
Processed: 5
```

### 步骤 2: 保持顺序

如果需要保持输入顺序，使用 `executor.map()`：

```python
with ThreadPoolExecutor(max_workers=3) as executor:
    # map 方法保持输入顺序
    results = list(executor.map(process_item, items))
    
for result in results:
    print(result)
```

### 步骤 3: 错误处理

```python
def process_item_with_error(item):
    if item == 3:
        raise ValueError(f"Error at {item}")
    return f"Processed: {item}"

with ThreadPoolExecutor(max_workers=3) as executor:
    futures = {executor.submit(process_item_with_error, item): item 
               for item in items}
    
    for future in as_completed(futures):
        try:
            result = future.result()
            print(f"Success: {result}")
        except Exception as e:
            item = futures[future]
            print(f"Failed {item}: {e}")
```

## 📋 核心 API

### 1. ThreadPoolExecutor 初始化

```python
from concurrent.futures import ThreadPoolExecutor

executor = ThreadPoolExecutor(
    max_workers=4,        # 最大线程数（默认：CPU核心数 * 5）
    thread_name_prefix='', # 线程名称前缀
    initializer=None,     # 线程初始化函数
    initargs=()           # 初始化函数参数
)
```

### 2. 提交任务

```python
# 方法 1: submit - 提交单个任务，返回 Future 对象
future = executor.submit(function, *args, **kwargs)

# 方法 2: map - 批量提交，保持顺序
results = executor.map(function, iterable, timeout=None, chunksize=1)
```

### 3. 获取结果

```python
# 方法 1: future.result() - 阻塞等待结果
result = future.result(timeout=None)

# 方法 2: as_completed() - 按完成顺序获取
from concurrent.futures import as_completed

for future in as_completed(futures):
    result = future.result()
```

### 4. Future 对象常用方法

```python
future.result(timeout=None)    # 获取结果（阻塞）
future.exception(timeout=None) # 获取异常
future.cancel()                # 取消任务
future.cancelled()             # 是否已取消
future.done()                  # 是否已完成
future.running()               # 是否正在运行
```

## 💡 常见场景

### 场景 1: 批量处理文件

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
import os

def process_file(file_path):
    """处理单个文件"""
    with open(file_path, 'r') as f:
        content = f.read()
    # 处理内容...
    return len(content)

# 获取所有文件
file_paths = [f"data/file_{i}.txt" for i in range(100)]

with ThreadPoolExecutor(max_workers=10) as executor:
    futures = {executor.submit(process_file, path): path 
               for path in file_paths}
    
    results = {}
    for future in as_completed(futures):
        file_path = futures[future]
        try:
            size = future.result()
            results[file_path] = size
        except Exception as e:
            print(f"Error processing {file_path}: {e}")
```

### 场景 2: 并发网络请求

```python
import requests
from concurrent.futures import ThreadPoolExecutor, as_completed

def fetch_url(url):
    """获取 URL 内容"""
    response = requests.get(url, timeout=10)
    return response.status_code, len(response.content)

urls = [
    "https://api.example.com/data1",
    "https://api.example.com/data2",
    # ... 更多 URL
]

with ThreadPoolExecutor(max_workers=5) as executor:
    futures = {executor.submit(fetch_url, url): url for url in urls}
    
    for future in as_completed(futures):
        url = futures[future]
        try:
            status, size = future.result()
            print(f"{url}: {status} ({size} bytes)")
        except Exception as e:
            print(f"{url}: Error - {e}")
```

### 场景 3: 带超时的任务

```python
from concurrent.futures import ThreadPoolExecutor, TimeoutError as FutureTimeoutError
import time

def long_running_task(duration):
    time.sleep(duration)
    return f"Completed in {duration}s"

with ThreadPoolExecutor(max_workers=3) as executor:
    future = executor.submit(long_running_task, 5)
    
    try:
        # 设置超时时间
        result = future.result(timeout=2)
        print(result)
    except FutureTimeoutError:
        print("Task timed out!")
        future.cancel()  # 尝试取消任务
```

### 场景 4: 使用 Generator 作为输入

```python
def data_generator():
    """生成器函数"""
    for i in range(100):
        yield i

# ThreadPoolExecutor 可以直接使用 generator
with ThreadPoolExecutor(max_workers=5) as executor:
    # map 方法支持 generator
    results = list(executor.map(process_item, data_generator()))
    
    # 或者使用 submit
    futures = [executor.submit(process_item, item) 
               for item in data_generator()]
```

```python
# I/O 密集型：可以设置较大值
max_workers = 20  # 或更多
```

### 2. 使用上下文管理器

```python
# ✅ 推荐：自动清理资源
with ThreadPoolExecutor(max_workers=4) as executor:
    futures = [executor.submit(task, item) for item in items]
    # 自动等待所有任务完成并关闭

# ❌ 不推荐：需要手动关闭
executor = ThreadPoolExecutor(max_workers=4)
futures = [executor.submit(task, item) for item in items]

```

### 3. 处理异常

```python
# ✅ 推荐：为每个 future 单独处理异常
futures = {executor.submit(task, item): item for item in items}
for future in as_completed(futures):
    try:
        result = future.result()
    except Exception as e:
        item = futures[future]
        print(f"Task {item} failed: {e}")
```

### 4. 避免共享可变状态

```python
# ❌ 不推荐：线程不安全
results = []
def task(item):
    results.append(item)  # 可能导致数据竞争

# ✅ 推荐：返回结果
def task(item):
    return item

with ThreadPoolExecutor() as executor:
    results = list(executor.map(task, items))
```

### 5. 使用 as_completed 获取实时结果

```python
# ✅ 推荐：实时获取完成的任务
for future in as_completed(futures):
    result = future.result()
    # 立即处理结果，不需要等待所有任务完成

# ⚠️ 注意：如果需要保持顺序，使用 executor.map()
results = list(executor.map(task, items))
```

## 🔥 ProcessPoolExecutor 操作指南

`ProcessPoolExecutor` 是 Python 标准库提供的进程池执行器，用于并行执行 CPU 密集型任务。

**适用场景：**
- ✅ CPU 密集型任务（数值计算、图像处理、数据压缩）
- ✅ 需要绕过 GIL（全局解释器锁）限制的任务
- ❌ I/O 密集型任务（应使用 `ThreadPoolExecutor`）

---

### 🚀 快速开始

#### 步骤 1: 基本用法

```python
from concurrent.futures import ProcessPoolExecutor, as_completed
import math

def compute_square(n):
    """CPU 密集型计算"""
    result = sum(math.sqrt(i) for i in range(n))
    return f"Sum of sqrt(0..{n-1}) = {result:.2f}"

# 准备数据
numbers = [1000000, 2000000, 3000000, 4000000, 5000000]

# 使用进程池执行
with ProcessPoolExecutor(max_workers=4) as executor:
    # 提交所有任务
    futures = [executor.submit(compute_square, num) for num in numbers]
    
    # 获取结果（按完成顺序）
    for future in as_completed(futures):
        result = future.result()
        print(result)
```

#### 步骤 2: 使用 map 方法

```python
with ProcessPoolExecutor(max_workers=4) as executor:
    # map 方法保持输入顺序
    results = list(executor.map(compute_square, numbers))
    
for result in results:
    print(result)
```

#### 步骤 3: 错误处理

```python
def compute_with_error(n):
    if n == 3000000:
        raise ValueError(f"Error at {n}")
    return sum(math.sqrt(i) for i in range(n))

with ProcessPoolExecutor(max_workers=4) as executor:
    futures = {executor.submit(compute_with_error, num): num 
               for num in numbers}
    
    for future in as_completed(futures):
        try:
            result = future.result()
            print(f"Success: {result}")
        except Exception as e:
            num = futures[future]
            print(f"Failed {num}: {e}")
```

### 📋 ProcessPoolExecutor 核心 API

#### 1. 初始化

```python
from concurrent.futures import ProcessPoolExecutor

executor = ProcessPoolExecutor(
    max_workers=None,      # 最大进程数（默认：CPU 核心数）
    mp_context=None,       # 多进程上下文（可选）
    initializer=None,      # 进程初始化函数
    initargs=()            # 初始化函数参数
)
```

**重要参数说明：**
- `max_workers`: 通常设置为 CPU 核心数（`os.cpu_count()`）
- `initializer`: 每个进程启动时执行的函数（用于初始化共享资源）

#### 2. 提交任务

```python
# 方法 1: submit - 提交单个任务
future = executor.submit(function, *args, **kwargs)

# 方法 2: map - 批量提交，保持顺序
results = executor.map(function, iterable, timeout=None, chunksize=1)
```

**注意：** `chunksize` 参数用于控制每个进程处理的数据块大小，可以优化性能。

#### 3. 获取结果

```python
# 方法 1: future.result() - 阻塞等待结果
result = future.result(timeout=None)

# 方法 2: as_completed() - 按完成顺序获取
from concurrent.futures import as_completed

for future in as_completed(futures):
    result = future.result()
```

### ⚠️ 重要注意事项

#### 1. 数据序列化要求

**所有传递给进程的数据必须可以序列化（pickle）：**

```python
# ✅ 可以序列化：基本类型、列表、字典、类实例
def process_data(data):
    return data * 2

with ProcessPoolExecutor() as executor:
    results = executor.map(process_data, [1, 2, 3, 4, 5])

# ❌ 不能序列化：lambda 函数、局部函数、文件对象
# lambda x: x * 2  # 不能直接使用
# 需要定义为普通函数
```

#### 2. 函数必须定义在模块级别

```python
# ✅ 正确：函数定义在模块级别
def process_item(item):
    return item * 2

# ❌ 错误：函数定义在类方法或嵌套函数中
class MyClass:
    def process(self, item):
        return item * 2  # 不能直接用于 ProcessPoolExecutor
```

**解决方案：** 将函数提取到模块级别，或使用 `functools.partial`：

```python
from functools import partial

class MyClass:
    def process(self, item):
        return item * 2

obj = MyClass()
# 使用 partial 包装
process_func = partial(MyClass.process, obj)

with ProcessPoolExecutor() as executor:
    results = executor.map(process_func, [1, 2, 3])
```

#### 3. Windows 平台特殊要求

**在 Windows 上，必须使用 `if __name__ == '__main__':` 保护：**

```python
from concurrent.futures import ProcessPoolExecutor

def process_item(item):
    return item * 2

if __name__ == '__main__':  # Windows 必需！
    with ProcessPoolExecutor(max_workers=4) as executor:
        results = list(executor.map(process_item, [1, 2, 3, 4, 5]))
        print(results)
```

#### 4. 内存开销

**每个进程都有独立的内存空间，注意内存使用：**

```python
# ⚠️ 注意：大量数据会复制到每个进程
large_data = [list(range(1000000)) for _ in range(10)]

# 如果数据很大，考虑：
# 1. 减少 max_workers
# 2. 使用共享内存（multiprocessing.shared_memory）
# 3. 传递文件路径而不是数据本身
```

#### 5. 进程启动开销

**进程创建比线程慢，适合长时间运行的任务：**

```python
# ✅ 适合：耗时较长的 CPU 密集型任务
def heavy_computation(n):
    return sum(i**2 for i in range(n))

# ❌ 不适合：快速任务（进程启动开销大于任务本身）
def quick_task(x):
    return x + 1
```

### 🆚 ThreadPoolExecutor vs ProcessPoolExecutor

| 特性 | ThreadPoolExecutor | ProcessPoolExecutor |
|------|-------------------|---------------------|
| **适用场景** | I/O 密集型 | CPU 密集型 |
| **内存** | 共享内存 | 独立内存空间 |
| **开销** | 较小 | 较大（进程创建） |
| **GIL 影响** | 受 GIL 限制 | 不受 GIL 限制 |
| **数据传递** | 直接传递对象 | 需要序列化（pickle） |

**选择建议：**
- 网络请求、文件 I/O → `ThreadPoolExecutor`
- 数值计算、图像处理 → `ProcessPoolExecutor`

## ❓ 常见问题

### Q1: 如何限制并发数量？

```python
# 使用 max_workers 参数
executor = ThreadPoolExecutor(max_workers=5)  # 最多 5 个并发线程
```

### Q2: 如何等待所有任务完成？

```python
# 方法 1: 使用上下文管理器（推荐）
with ThreadPoolExecutor() as executor:
    futures = [executor.submit(task, item) for item in items]
    # 退出 with 块时自动等待

# 方法 2: 手动等待
for future in futures:
    future.result()  # 阻塞等待
```

### Q3: 如何取消任务？

```python
future = executor.submit(task, item)
future.cancel()  # 只能取消未开始的任务

# 检查是否取消成功
if future.cancelled():
    print("Task was cancelled")
```

### Q4: Generator 可以直接使用吗？

```python
# ✅ 可以！ThreadPoolExecutor 支持 generator
def data_gen():
    for i in range(100):
        yield i

with ThreadPoolExecutor() as executor:
    # map 方法支持 generator
    results = list(executor.map(process_item, data_gen()))
```

### Q5: 如何获取任务进度？

```python
from concurrent.futures import as_completed

total = len(items)
completed = 0

with ThreadPoolExecutor() as executor:
    futures = {executor.submit(task, item): item for item in items}
    
    for future in as_completed(futures):
        completed += 1
        progress = (completed / total) * 100
        print(f"Progress: {completed}/{total} ({progress:.1f}%)")
        result = future.result()
```

### Q6: 任务超时怎么办？

```python
from concurrent.futures import TimeoutError as FutureTimeoutError

future = executor.submit(long_task, item)

try:
    result = future.result(timeout=5)  # 5 秒超时
except FutureTimeoutError:
    print("Task timed out")
    future.cancel()
```
